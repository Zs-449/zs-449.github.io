1. System Architecture

A lightweight Client-Server model tailored for student projects. Front-end pages request data from a Node.js REST API using the fetch() API. Authentication uses session/JWT validation. Data is stored persistently in a relational MySQL database.

4. REST API Endpoints

POST /api/auth/login

Authenticates users & returns a session token or user role object.

POST /api/appointments

Books a new appointment. Checks for double-booking conflicts.

GET /api/appointments/:id

Retrieves appointments mapped to a user ID or role filter.

PUT /api/appointments/reschedule

Updates the appointment date & time slot.

5. Sample Backend Code (Node.js/Express)

const express = require('express');
const mysql = require('mysql2');
const app = express();

app.use(express.json());

// MySQL Database Connection Pool
const pool = mysql.createPool({
    host: 'localhost',
    user: 'root',
    password: '',
    database: 'upnm_clinic'
});

// Endpoint: Book Appointment
app.post('/api/appointments', (req, res) => {
    const { patientId, doctorId, date, timeSlot, illnessType } = req.body;

    // 1. Double Booking Check Query
    const checkQuery = `SELECT * FROM appointments WHERE doctor_id = ? AND date = ? AND time_slot = ?`;
    
    pool.query(checkQuery, [doctorId, date, timeSlot], (err, results) => {
        if (err) return res.status(500).json({ error: err.message });
        if (results.length > 0) {
            return res.status(409).json({ message: "Time slot already booked!" });
        }

        // 2. Insert Booking
        const insertQuery = `INSERT INTO appointments (patient_id, doctor_id, date, time_slot, status) VALUES (?, ?, ?, ?, 'scheduled')`;
        pool.query(insertQuery, [patientId, doctorId, date, timeSlot], (err, result) => {
            if (err) return res.status(500).json({ error: err.message });
            res.status(201).json({ message: "Booking successful!", id: result.insertId });
        });
    });
});

app.listen(3000, () => console.log('UPNM Clinic Server running on port 3000'));
